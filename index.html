} else { p.textContent = label; }
p.onclick = () => { activeRole = id; updateFilters(); };
rf.appendChild(p);
});
const cf = document.getElementById('cat-filters');
[{id:'all',label:'Alle'},...Object.entries(CAT_LABELS).map(([id,label])=>({id,label}))].forEach(c => {
const p = document.createElement('span');
p.className = 'filter-pill' + (c.id === 'all' ? ' active' : '');
p.dataset.c = c.id;
p.textContent = c.label;
p.onclick = () => { activeCat = c.id; updateFilters(); };
cf.appendChild(p);
});
}

function updateFilters() {
document.querySelectorAll('[data-r]').forEach(p => p.classList.toggle('active', p.dataset.r === activeRole));
document.querySelectorAll('.filter-pill[data-c]').forEach(p => p.classList.toggle('active', p.dataset.c === activeCat));
let count = 0;
document.querySelectorAll('.entry').forEach(e => {
const matchCat = activeCat === 'all' || e.dataset.c === activeCat;
const matchRole = activeRole === 'all' || e.dataset.roles.includes(activeRole);
const show = matchCat && matchRole;
e.classList.toggle('hidden', !show);
if (!show) e.classList.remove('open');
if (show) count++;
});
document.getElementById('counter').innerHTML = `<b>${count}</b> von 56 Einträgen sichtbar`;
}
